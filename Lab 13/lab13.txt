CREATE OR REPLACE TRIGGER keep_minmax_salary
    BEFORE UPDATE ON jobs
    FOR EACH ROW
    DECLARE
        PRAGMA AUTONOMOUS_TRANSACTION;
    BEGIN
        IF NOT :NEW.min_salary = :OLD.min_salary THEN
            :NEW.min_salary := :OLD.min_salary;
        END IF;
        IF NOT :NEW.max_salary = :OLD.max_salary THEN
            :NEW.max_salary := :OLD.max_salary;
        END IF;
        COMMIT;
    END;

INSERT INTO jobs VALUES('ID_TEST', 'Test', 2000, 4000);

UPDATE jobs SET min_salary = 3000 WHERE job_id = 'ID_TEST';
UPDATE jobs SET max_salary = 6000 WHERE job_id = 'ID_TEST';
